Thuộc tính :active Pseudo-class trong CSS


a:link { color: blue; } /* màu link */
a:hover { background: yellow; } /* khi rê chuột vào */
a:visited { color: purple; } /* Đổi màu sau khi click vào link, những link đã click sẻ khác màu */
a:active { color: red; } /* Khi nhấn giữ */
p:active { background: gray; color:#fff; } /* Khi nhấn giữ đoạn văn bản thì đổi màu nền */
a:focus { color:red} /* Nhấn giữ và rê chuột thì đổi màu, khi nào click chuột lại thì mới mất */
